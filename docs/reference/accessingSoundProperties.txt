    if (secondSound && secondSound.isValid) {
        var secondSoundVolume = secondSound._native.volume;         // e.g., -6

    // Get count of sounds in container

// 3. CREATING NEW SOUNDS WITH COPIED PROPERTIES
function createNewSingleSound(original, audioFilePath) {
    // Create a new SingleSound
    var newSound = studio.project.create('SingleSound');
    
    // Set audio file
    newSound.audioFile = studio.project.workspace.masterAssetFolder.getAsset(audioFilePath);
    
    // Copy properties from original
    newSound.volume = original.volume;
    newSound.pitch = original.pitch;
    newSound.startOffset = original.startOffset;
    newSound.looping = original.looping;
    newSound.playCount = original.playCount;
    newSound.triggerProbability = original.triggerProbability;
    newSound.isAsync = original.isAsync;
    newSound.voiceStealing = original.voiceStealing;
    newSound.maxVoices = original.maxVoices;
    
    return newSound;
}

// 4. CREATING NEW MULTISOUND WITH COPIED PROPERTIES
function createNewMultiSound(original, audioFilePaths) {
    // Create a new MultiSound
    var newMulti = studio.project.create('MultiSound');
    
    // Copy container properties
    newMulti.playlistMode = original.playlistMode;
    newMulti.looping = original.looping;
    
    // Create and add sounds
    for (var i = 0; i < original.sounds.length; i++) {
        if (original.sounds[i] && original.sounds[i].isValid) {
            // Create a sound of the same type
            var newSound = studio.project.create(original.sounds[i].entity);
            
            // Set audio file if provided, otherwise keep original
            if (audioFilePaths && audioFilePaths[i]) {
                newSound.audioFile = studio.project.workspace.masterAssetFolder.getAsset(audioFilePaths[i]);
            } else if (original.sounds[i].audioFile) {
                newSound.audioFile = original.sounds[i].audioFile;
            }
            
            // Copy properties from _native when available
            if (original.sounds[i]._native) {
                newSound.volume = original.sounds[i]._native.volume;
                newSound.pitch = original.sounds[i]._native.pitch;
                newSound.startOffset = original.sounds[i]._native.startOffset;
            } else {
                // Fall back to direct properties
                newSound.volume = original.sounds[i].volume;
                newSound.pitch = original.sounds[i].pitch;
                newSound.startOffset = original.sounds[i].startOffset;
            }
            
            // Set owner to add to the MultiSound
            newSound.owner = newMulti;
        }
    }
    
    return newMulti;
}
